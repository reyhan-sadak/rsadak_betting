CREATE TABLE Users
(
ID INT NOT NULL AUTO_INCREMENT,
PRIMARY KEY(ID),
Name CHAR(31),
Email CHAR(31),
PasswordHash CHAR(128),
UserHash CHAR(128),
UserRank INT NOT NULL,
UpdatedTime TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

CREATE TABLE FootballLeagues
(
ID INT NOT NULL AUTO_INCREMENT,
PRIMARY KEY(ID),
Name CHAR(31),
CreatorId INT NOT NULL,
UpdatedTime TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

CREATE TABLE FootballTeams
(
ID INT NOT NULL AUTO_INCREMENT,
PRIMARY KEY(ID),
Name CHAR(31),
LeagueId INT NOT NULL,
CreatorId INT NOT NULL,
UpdatedTime TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

CREATE TABLE MatchGroups
(
ID INT NOT NULL AUTO_INCREMENT,
PRIMARY KEY(ID),
Name CHAR(31),
FromDate DATE,
ToDate DATE,
CreatorId INT NOT NULL,
UpdatedTime TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

CREATE TABLE FootballMatches
(
ID INT NOT NULL AUTO_INCREMENT,
PRIMARY KEY(ID),
HostTeamId INT NOT NULL,
GuestTeamId INT NOT NULL,
DateAndTime DATETIME,
HostScore INT,
GuestScore INT,
MatchGroupId INT NOT NULL,
CreatorId INT NOT NULL,
UpdatedTime TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

CREATE TABLE FootballMatchPredictions
(
ID INT NOT NULL AUTO_INCREMENT,
PRIMARY KEY(ID),
MatchId INT NOT NULL,
HostScore INT,
GuestScore INT,
CreatorId INT NOT NULL,
UpdatedTime TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);
